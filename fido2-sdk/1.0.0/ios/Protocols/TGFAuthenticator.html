TGFAuthenticator

@protocol TGFAuthenticator <NSObject>

- (instancetype)init NS_UNAVAILABLE;

/**
 Indicates if a Authenticator is enrolled.
 Use this to determine if the user needs to be enrolled.
 
 @return YES if enrolled.
 */
-(BOOL)isEnrolled;

/**
 Enroll a Authenticator. Users MUST be enrolled to the specific authenticator
 to perform FIDO2 operations. The enrollment process stores a secure means of identification
 which will be used to verify the user when authentication is required.
 */
-(void)enroll;

/**
 Unenroll a previously enrolled Authenticator.
 User authentication is required prior to a successful unenrollment.
 Upon successful unenrollment, all FIDO2 registrations for the specific authenticator will be invalidated.
 A subsequent enrollment is required to once again perform FIDO2 operations.
 */
-(void)unenroll;


@end
