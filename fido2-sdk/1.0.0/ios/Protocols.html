    @protocol TGFAuthenticator <NSObject>

- (instancetype)init NS_UNAVAILABLE;

/**
 Indicates if a Authenticator is enrolled.
 Use this to determine if the user needs to be enrolled.
 
 @return YES if enrolled.
 */
-(BOOL)isEnrolled;

/**
 Enroll a Authenticator. Users MUST be enrolled to the specific authenticator
 to perform FIDO2 operations. The enrollment process stores a secure means of identification
 which will be used to verify the user when authentication is required.
 */
-(void)enroll;

/**
 Unenroll a previously enrolled Authenticator.
 User authentication is required prior to a successful unenrollment.
 Upon successful unenrollment, all FIDO2 registrations for the specific authenticator will be invalidated.
 A subsequent enrollment is required to once again perform FIDO2 operations.
 */
-(void)unenroll;


@end

    @protocol TGFFido2Client <NSObject>

/**
 Responds asynchronously to a FIDO2 request by generating a FIDO2
 response. This operation may run for a long period of time and all UI is
 delegated to the caller.
 
 @param respondArgs FIDO2 request plus other arguments required to
 fulfill the request.
 @param completionBlock The completion handler to call when the execution is complete.
    It takes the following parameters:
    * response - Success when it is not nil.
    * error - Error details. See `TGFFido2Error` for error types.
 
@since 1.0.0
 */
- (id<TGFFido2Task> _Nullable)respondWithArgs:(TGFFido2RespondArgs*_Nonnull)respondArgs
             completion:(void(^)(id<TGFFido2Response> _Nullable response, NSError*_Nullable error))completionBlock;

/**
 Retrieve a list of registered FIDO2 Authenticators.
 
 @return An array of TGFFido2AuthenticatorInfo representing the registered Authenticators
 
 @since 1.0.0
 */
- (NSArray<TGFFido2AuthenticatorRegistrationInfo *> *)authenticatorRegistrations;


/**
Delete Authenticator registration

@since 1.0.0
*/
- (void)deleteAuthenticatorRegistration:(TGFFido2AuthenticatorRegistrationInfo*)authenticatorRegistration;

/**
 Delete all authenticator registrations.

 @note This method will also delete Passcode Authenticator enrollment.

 @since 1.0.0
 */
- (void)deleteAuthenticatorRegistrations;

@end
